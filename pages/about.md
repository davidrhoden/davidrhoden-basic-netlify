<div class="about-intro">
  <p>David Rhoden is a painter, animator, and illustrator, currently based in New Orleans. And I'm David Rhoden.</p>

  <p>Not only do I paint, illustrate, and animate, I also work for a living.</p>

  <p>I have had more jobs than you can count, including being an ice cream man and an oyster shucker for one day each. I started my post-collegiate work career as a freelance writer and an assistant editor at Golden Books, spent some years toiling in various departments at Citibank and Goldman Sachs, then, through a series of temp jobs, found my way into the business of design and marketing, where I work as a web developer and (infrequently) designer.</p>

  <p>I am also a musician, currently playing guitar with The Stacks; previously, I was a member of Mountain Shoutin', Jimmy and the Wolfpack, All-Night Movers, The Sleepy Heads, MAXIMVS!, Dirty Knives, The Hi-Balls, Snake Oil, Everything Tool, The Innuendos, Dark Marbles, Cheerio and the Camels, and The Red M&Ms.</p>
</div>

<script>
(function () {
  var carousel = document.getElementById('about-gallery-carousel');
  if (!carousel) return;
  if (window.matchMedia('(max-width: 959px)').matches) return;
  var gallery = carousel.querySelector('.about-gallery');
  var prevBtn = carousel.querySelector('.carousel-btn-prev');
  var nextBtn = carousel.querySelector('.carousel-btn-next');

  function getSlides() {
    return Array.prototype.slice.call(gallery.querySelectorAll('.about-slide'));
  }

  function getCurrentIndex() {
    var slides = getSlides();
    var scrollLeft = gallery.scrollLeft;
    var closest = 0;
    var minDiff = Infinity;
    slides.forEach(function (slide, i) {
      var diff = Math.abs(slide.offsetLeft - scrollLeft);
      if (diff < minDiff) {
        minDiff = diff;
        closest = i;
      }
    });
    return closest;
  }

  function scrollToSlide(index) {
    var slides = getSlides();
    if (index < 0) index = 0;
    if (index >= slides.length) index = slides.length - 1;
    gallery.scrollTo({ left: slides[index].offsetLeft, behavior: 'smooth' });
  }

  if (prevBtn) {
    prevBtn.addEventListener('click', function () {
      scrollToSlide(getCurrentIndex() - 1);
    });
  }
  if (nextBtn) {
    nextBtn.addEventListener('click', function () {
      scrollToSlide(getCurrentIndex() + 1);
    });
  }

  gallery.addEventListener('wheel', function (e) {
    if (Math.abs(e.deltaY) > Math.abs(e.deltaX)) {
      var atStart = gallery.scrollLeft <= 0;
      var atEnd = gallery.scrollLeft >= gallery.scrollWidth - gallery.clientWidth - 1;
      var goingRight = e.deltaY > 0;
      if ((goingRight && !atEnd) || (!goingRight && !atStart)) {
        e.preventDefault();
        scrollToSlide(getCurrentIndex() + (goingRight ? 1 : -1));
      }
    }
  }, { passive: false });
}());
</script>
